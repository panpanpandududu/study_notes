#### 根据字符串来访问对象的层级结构

```js
var obj={
    a:{
        b:{
            c:100
        }
    }
};
Object.prototype.getValueByStr=function(str){
    if（str.indeOf（'.'）==-1） return this[str]；
    let res=this;
    let arr=str.split('.')；
    for(let i=0;i<arr.length;i++){
        res = res[arr[i]];
        if(!res) return res;
        if(i==arr.length-1)  return res;
    }
}
obj.getValueByStr('a.b.c');
```
